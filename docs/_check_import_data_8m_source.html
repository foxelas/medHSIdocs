27datadir = fullfile(config.GetSetting('dataDir'), config.GetSetting('dataFolderName'), '*.h5');
28v = dir(datadir);
29vname = {v.name};
30k = cellfun(@(x) strrep(x, '.hsm', ''), vname, 'UniformOutput', false);
31fileS = struct('Filename', [], 'SampleID', [], 'Target', [], 'Content', [], ...
32 'ROI', [], 'IntegrationTime', [], 'CaptureDate', [], 'IsUnfixed', [], 'IsBackside', []);
33for i = 1:numel(k)
34 flag = false;
35 name = k{i};
36 parts = strsplit(name, '_');
37 fileS(i).Filename = name;
38 if numel(parts) < 3
39 fileS(i).SampleID = '';
40 flag = true;
41 else
42 fileS(i).SampleID = parts{3};
43 end
44
45 if contains(name, 'black')
46 fileS(i).Content = 'black';
47 elseif contains(name, 'white')
48 fileS(i).Content = 'white';
49 elseif contains(name, 'row') || contains(name, 'raw') || contains(name, 'fix')
50 fileS(i).Content = 'tissue';
51 else
52 fileS(i).Content = '';
53 end
54 if contains(name, 'fix')
55 fileS(i).Target = strcat(fileS(i).SampleID, '_', 'fix');
56 fileS(i).IsUnfixed = 0;
57 else
58 fileS(i).Target = strcat(fileS(i).SampleID, '_', 'raw');
59 fileS(i).IsUnfixed = 1;
60 end
61
62 fileS(i).IsBackside = isempty(str2double(fileS(i).SampleID));
63
64 flag = flag || contains(name, 'row');
65 flag = flag || ~contains(name, 'raw') && ~contains(name, 'fix');
66
67 if flag
68 fprintf('Error at %s \n', name);
69 end
70
71 fileS(i).IntegrationTime = 618;
72 fileS(i).CaptureDate = parts{1};
73end
74
75%% Check part
76disp('Check Starts')
77flag = true;
78for i = 1:numel(k)
79 if strcmp(fileS(i).Content, 'tissue')
80 target = fileS(i).Target;
81 if sum(strcmp({fileS.Target}, target)) > 3
82 flag = false;
83 fprintf('Error [More than 3 images] at %s \n', fileS(i).Filename);
84 else
85 b = sum(strcmp({fileS.Target}, target) & strcmp({fileS.Content}, 'black')) == 1;
86 w = sum(strcmp({fileS.Target}, target) & strcmp({fileS.Content}, 'white')) == 1;
87 if ~(b && w) && ~contains(target, 'b')
88 flag = false;
89 fprintf('Error [No white or black] at %s \n', fileS(i).Filename);
90 end
91 end
92
93 end
94
95 if strcmp(fileS(i).Content, '')

97 fprintf('Error [No Content] at %s \n', fileS(i).Filename);
98 end
99end
100

106 end
107end
