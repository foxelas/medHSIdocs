24% GetDisplayImageInternal returns an RGB image from the hyperspectral data.
25%
26% Parameters are saved in config::[paramDir]\[parametersFileName]
27%
28% @b Usage
29%
30% @code
31% dispImage = GetDisplayImageInternal(hsIm, superixelNumber);
32%
33% dispImage = GetDisplayImageInternal(hsIm, 'rgb');
34%
35% dispImage = GetDisplayImageInternal(hsIm, 'channel', 200);
36% @endcode
37%
38% @param obj [hsi] | An instance of the hsi class
39% @param method [string] | The method for display image creation
40% ('rgb' or 'channel')
41% @param channelNumber [int] | The target channel number
42%
43% @retval dispImage [numeric array] | The display image
44
45if nargin < 2
46 method = 'rgb';
47end
48
49if nargin < 3
50 channel = 100;
51end
52
53[m, n, z] = size(hsIm);
54if (z < 401)
55 v = hsiUtility.GetWavelengths(z, 'index');
56 spectralImage2 = zeros(m, n, 401);
57 spectralImage2(:, :, v) = hsIm;
58 hsIm = spectralImage2;
59 z = 401;
60 clear 'spectralImage2';
61end
62if config.HasGPU()
63 spectralImage_ = gpuArray(hsIm);
64else
65 spectralImage_ = hsIm;
66end
67clear 'spectralImage';
68
69switch method
70 case 'rgb'
71 %[lambda, xFcn, yFcn, zFcn] = colorMatchFcn('CIE_1964');
72 colImage = double(reshape(spectralImage_, [m * n, z]));
73
74 [xyz, illumination] = PrepareParams(z);
75 normConst = double(max(max(colImage)));
76 colImage = colImage ./ normConst;
77 colImage = bsxfun(@times, colImage, illumination);
78 colXYZ = colImage * squeeze(xyz);
79 clear 'colImage';
80
81 imXYZ = reshape(colXYZ, [m, n, 3]);
82 imXYZ = max(imXYZ, 0);
83 imXYZ = imXYZ / max(imXYZ(:));
84 dispImage_ = XYZ2sRGB_exgamma(imXYZ);
85 dispImage_ = max(dispImage_, 0);
86 dispImage_ = min(dispImage_, 1);
87 dispImage_ = dispImage_.^0.4;
88
89 case 'channel'
90 dispImage_ = rescale(spectralImage_(:, :, channel));
91 otherwise
92 error('Unsupported method for display image reconstruction');
93end
94
95if config.HasGPU()
96 dispImage = gather(dispImage_);
97else
98 dispImage = dispImage_;
99end
100
101end

103function [xyz, illumination] = PrepareParams(z)
104filename = commonUtility.GetFilename('param', config.GetSetting('parametersFileName'));
105if ~exist(filename, 'file')
106 lambdaIn = hsiUtility.GetWavelengths(z, 'raw');
107 [lambdaMatch, xFcn, yFcn, zFcn] = colorMatchFcn('1964_FULL');
108 xyz = interp1(lambdaMatch', [xFcn; yFcn; zFcn]', lambdaIn, 'pchip', 0);
109 [solaxSpec, lambdaMatch] = GetSolaxSpectra();
110 illumination = interp1(lambdaMatch, solaxSpec', lambdaIn, 'pchip', 0);
111 save(filename, 'xyz', 'illumination');
112else
113 load(filename, 'xyz', 'illumination');
114end
115end
