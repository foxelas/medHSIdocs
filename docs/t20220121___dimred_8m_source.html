241function [] = plotTwoAxes(scoresVals, labels, methodName)
242healthy = find(labels == 0);
243tumor = find(labels == 1);
244
245hold on;
246if size(scoresVals, 2) > 1
247 scatter(scoresVals(healthy, 1), scoresVals(healthy, 2), 'go');
248 scatter(scoresVals(tumor, 1), scoresVals(tumor, 2), 'rd');
249else
250 scatter(1:length(healthy), scoresVals(healthy), 'go');
251 scatter(1:length(tumor), scoresVals(tumor), 'rd');
252end
253hold off;
254
255title(methodName)
256if size(scoresVals, 2) > 1
257 xlabel('Projected Axis 1');
258 ylabel('Projected Axis 2');
259else
260 ylabel('Projected Axis');
261 xlabel('Samples');
262end
263end
