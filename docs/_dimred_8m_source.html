40% Dimred reduces the dimensions of the hyperspectral image.
41%
42% Currently PCA, ICA (FastICA), RICA, SuperPCA, MSuperPCA, LDA, QDA, Wavelength-Selection are available.
43% Additionally, for pre-trained parameters RFI and Autoencoder are available.
44% For an unknown method, the input data is returned.
45% For more details check @c function Dimred.
46%
47% @b Usage
48%
49% @code
50% q = 10;
51% [coeff, scores, latent, explained, objective] = Dimred(X,
52% method, q, mask);
53%
54% [coeff, scores, latent, explained, ~] = Dimred(X, 'pca', 10);
55%
56% [coeff, scores, ~, ~, objective] = Dimred(X, 'rica', 40);
57%
58% [~, scores, ~, ~, ~, Mdl] = Dimred(X, 'lda')
59% @endcode
60%
61% @param X [numeric array] | The input data as a matrix with MxN observations and Z columns
62% @param method [string] | The method for dimension reduction
63% @param q [int] | The number of components to be retained
64% @param mask [numerical array] | A 2D logical array marking pixels to be used in PCA calculation
65% @param varargin [cell array] | Optional additional arguments for methods that require them
66%
67% @retval coeff [numeric array] | The transformation coefficients
68% @retval scores [numeric array] | The transformed values
69% @retval latent [numeric array] | The latent values
70% @retval explained [numeric array] | The percentage of explained
71% variance
72% @retval objective [numeric array] | The objective function
73% values
74% @retval Mdl [model] | The dimension reduction model
75
76latent = [];
77explained = [];
78objective = [];
79Mdl = [];
80coeff = [];
81
82if nargin < 3
83 q = 10;
84end
85
86if nargin < 4
87 mask = [];
88end
89
90rng default % For reproducibility
91
92switch lower(method)
93%% PCA
94 case 'pca'
95 [coeff, scores, latent, ~, explained] = pca(X, 'NumComponents', q);
96
97%% FastICA
98 case 'ica'
99 [scores, ~, coeff] = fastica(X', 'numOfIC', q);
100
101%% RICA
102 case 'rica'
103 Mdl = rica(X, q, 'IterationLimit', 100, 'Lambda', 1);
104 coeff = Mdl.TransformWeights;
105 scores = X * coeff;
106 objective = Mdl.FitInfo.Objective;
107
108%% Discriminant Analysis (LDA / QDA)
109 case 'lda'
110 if isempty(mask)
111 error('A supervised method requires labels as argument');
112 end
113 Mdl = fitcdiscr(X, mask);
114 scores = predict(Mdl, X);
115
116 case 'qda'
117 if isempty(mask)
118 error('A supervised method requires labels as argument');
119 end
120 Mdl = fitcdiscr(X, mask, 'DiscrimType', 'quadratic');
121 scores = predict(Mdl, X);
122
123%% Wavelength Selection
124 case 'wavelength-selection'
125 wavelengths = hsiUtility.GetWavelengths(311);
126 id1 = find(wavelengths == 540);
127 id2 = find(wavelengths == 650);
128 scores = X(:, [id1, id2]);
129 coeff = zeros(wavelegths);
130 coeff(id1) = 1;
131 coeff(id2) = 1;
132
133%% SuperPCA
134 case 'superpca'
135 if isempty(varargin)
136 pixelNum = 20;
137 else
138 pixelNum = varargin{1};
139 end
140
141 %%super-pixels segmentation
142 superpixels = cubseg(X, pixelNum);
143
144 %%SupePCA based DR
145 scores = SuperPCA(X, q, superpixels);
146
147%% Multiscale SuperPCA
148 case 'msuperpca'
149
150 if isempty(varargin)
151 pixelNumArray = floor(20*sqrt(2).^[-2:2]);
152 else
153 pixelNumArray = varargin{1};
154 end
155
156 N = numel(pixelNumArray);
157 scores = cell(N, 1);
158 for i = 1:N
159 pixelNum = pixelNumArray(i);
160
161 %%super-pixels segmentation
162 superpixels = cubseg(X, pixelNum);
163
164 %%SupePCA based DR
165 scores{i} = SuperPCA(X, q, superpixels);
166 end
167
168 case 'rfi'
169 impOOB = varargin{1};
170 [~, idxOrder] = sort(impOOB, 'descend');
171 ido = idxOrder(1:q);
172 scores = X(:, ido);
173
174 case 'autoencoder'
175 autoenc = varargin{1};
176 scores = encode(autoenc, X')';
177
178 otherwise
179 scores = X;
180end
181
182
183end
