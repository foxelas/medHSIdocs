28% ReadTripletInternal reads and saves the three hyperspectral images
29%
30% The hyperspectral data are saved in .h5 format. The raw, white and black
31% (if exist) images are read one-by-one for the same target. Each HSI is
32% saved in config::[matDir]\[database]\[tripletsName]\*_xxx.mat, where
33% xxx is either '_target', '_white' or '_black'.
34%
35% To chose a mask for uni spectrum normalization, set config::['useCustomMask']
36%
37% @b Usage
38%
39% @code
40% content = 'tissue';
41% target = '001_raw';
42% spectralData = ReadTripletInternal(content, target);
43%
44% spectralData = ReadTripletInternal(content, target, blackIsCapOn);
45% @endcode
46%
47% @param content [cell array] | Contains the content to be imported
48% @param target [char] | Contains the target to be imported
49% @param blackIsCapOn [logical] | Flag about the use of blackCap for dark image
50%
51% @retval spectralData [numeric array] | The target (tissue) hyperspectral image
52
53if nargin < 3
54 blackIsCapOn = false;
55end
56
57%% Target image
58fcTarget = databaseUtility.GetFileConditions(content, target);
59[filename, tableId, ~] = databaseUtility.Query(fcTarget{:});
60if iscell(filename)
61 filename = filename{1};
62end
63targetName = num2str(tableId);
64[spectralData, ~, ~] = hsiUtility.ReadH5(filename);
65
66if ~exist(commonUtility.GetFilename('target', targetName), 'file') ...
67 || ~exist(commonUtility.GetFilename('black', targetName), 'file') ...
68 || ~exist(commonUtility.GetFilename('white', targetName), 'file')
69
70 plotBaseDir = commonUtility.GetFilename('output', fullfile(config.GetSetting('snapshotsFolderName'), 'ReadTriplets', targetName), '');
71
72 figure(1);
73 title('Target Image');
74 imshow(GetDisplayImageInternal(spectralData, 'rgb'));
75 config.SetSetting('plotName', config.DirMake(plotBaseDir, strcat(target, '_', num2str(config.GetSetting('integrationTime')))));
76 plots.SavePlot(1);
77 filename = config.DirMake(commonUtility.GetFilename('target', targetName));
78 fprintf('Target data [spectralData] is saved at %s.\n', filename);
79 save(filename, 'spectralData', '-v7.3');
80
81 if ~strcmp(config.GetSetting('normalization'), 'raw')
82
83 %% White image
84 if config.GetSetting('isTest')
85 fcWhite = databaseUtility.GetFileConditions('whiteReflectance', target);
86 else
87 fcWhite = databaseUtility.GetFileConditions('white', target);
88 end
89 [filename, ~, ~] = databaseUtility.Query(fcWhite{:});
90 if iscell(filename)
91 filename = filename{1};
92 end
93 [white, ~, wavelengths] = hsiUtility.ReadH5(filename);
94 figure(2);
95 title('White Reference Image');
96 imshow(GetDisplayImageInternal(white, 'rgb'));
97 config.SetSetting('plotName', config.DirMake(plotBaseDir, strcat('0_white_', num2str(config.GetSetting('integrationTime')))));
98 SavePlot(2);
99
100 %%UniSpectrum
101 if config.GetSetting('useCustomMask')
102 uniMask = GetCustomMaskInternal(white);
103 else
104 uniMask = ones(size(white, 1), size(white, 2));
105 end
106 uniSpectrum = GetMaskedPixelsInternal(white, uniMask);
107 config.SetSetting('plotName', config.DirMake(plotBaseDir, strcat('0_white_unispectrum_', num2str(config.GetSetting('integrationTime')))));
108 plots.Spectra(4, uniSpectrum, wavelengths, '99%-white', 'Reflectance Spectrum of White Balance Sheet');
109
110 %%BandMax
111 [m, n, w] = size(white);
112 bandmaxSpectrum = max(reshape(white, m*n, w), [], 1);
113 config.SetSetting('plotName', config.DirMake(plotBaseDir, strcat('0_white_bandmax_', num2str(config.GetSetting('integrationTime')))));
114 plots.Spectra(5, bandmaxSpectrum, wavelengths, 'Bandmax spectrum', 'Bandmax Spectrum for the current Image');
115
116 fullReflectanceByPixel = white;
117 filename = commonUtility.GetFilename('white', targetName);
118 fprintf('White data [fullReflectanceByPixel] is saved at %s.\n', filename);
119 save(filename, 'fullReflectanceByPixel', 'uniSpectrum', 'bandmaxSpectrum', '-v7.3');
120
121 %% Black Image
122 if config.GetSetting('isTest')
123 if blackIsCapOn
124 fcBlack = databaseUtility.GetFileConditions('capOn', target);
125 else
126 fcBlack = databaseUtility.GetFileConditions('lightsOff', target);
127 end
128 else
129 fcBlack = databaseUtility.GetFileConditions('black', target);
130 end
131 [filename, ~, ~] = databaseUtility.Query(fcBlack{:});
132 if iscell(filename)
133 filename = filename{1};
134 end
135 [blackReflectance, ~, ~] = hsiUtility.ReadH5(filename);
136 figure(3);
137 title('Black Reference Image');
138 imshow(GetDisplayImageInternal(blackReflectance, 'rgb'));
139 config.SetSetting('plotName', config.DirMake(plotBaseDir, strcat('0_black_', num2str(config.GetSetting('integrationTime')))));
140 plots.SavePlot(3);
141
142 filename = commonUtility.GetFilename('black', targetName);
143 fprintf('Black data [blackReflectance] is saved at %s.\n', filename);
144 save(filename, 'blackReflectance', '-v7.3');
145 else
146 disp('Read only capture data, ignore white and black images.');
147 end
148else
149 disp('Triple files already exist.');
150end
151
152end
