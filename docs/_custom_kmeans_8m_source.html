21% CustomKmeans applies kmeans clustering to an hsi and visualizes
22% the result.
23%
24% Need to set config::[saveFolder] for image output.
25%
26% @b Usage
27%
28% @code
29% CustomKmeans(hsIm, 5);
30%
31% apply.ToEach(@CustomKmeans, 5);
32% @endcode
33%
34% @param hsIm [hsi] | An instance of the hsi class
35% @param clusterNum [int] | The number of clusters
36%
37% @retval labels [numeric array] | The cluster labels
38
39srgb = hsIm.GetDisplayImage('rgb');
40fgMask = hsIm.FgMask;
41
42Xcol = hsIm.GetMaskedPixels(fgMask);
43[labelsCol, C] = kmeans(Xcol, clusterNum);
44
45labels = hsi.RecoverSpatialDimensions(labelsCol, size(fgMask), fgMask);
46
47savedir = commonUtility.GetFilename('output', fullfile(config.GetSetting('saveFolder'), config.GetSetting('fileName')), '');
48config.SetSetting('plotName', fullfile(savedir, 'kmeans-clustering'));
49plots.Superpixels(1, srgb, labels, '', 'color', fgMask);
50
51names = cell(clusterNum, 1);
52for i = 1:clusterNum
53 names{i} = strcat('Centroid', num2str(i));
54end
55config.SetSetting('plotName', fullfile(savedir, 'kmeans-centroids'));
56plots.Spectra(2, C, hsiUtility.GetWavelengths(size(hsIm.Value, 3)), names, 'Kmeans centroids');
57ylim([0, 1]);
58plots.SavePlot(2);
59end
