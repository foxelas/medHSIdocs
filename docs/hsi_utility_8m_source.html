63 %> @brief GetWavelengths returns the wavelengths of the hyperspectral image
64 %>
65 %> Depending on the option, it either returns the wavelengths or the
66 %> wavelength indexes of the spectral image.
67 %>
68 %> @b Usage
69 %>
70 %> @code
71 %> x = hsiUtility.GetWavelengths(m) returns wavelengths as a vector of wavelengths
72 %>
73 %> x = hsiUtility.GetWavelengths(m, 'raw') returns wavelengths as a vector of wavelengths
74 %>
75 %> x = hsiUtility.GetWavelengths(m, 'index') returns indexes respective to selected wavelengths
76 %>
77 %> x = hsiUtility.GetWavelengths(m, 'babel') returns indexes respective to selected wavelengths for babel standard spectra
78 %> @endcode
79 %>
80 %> @param m [int] | The total number of spectral channels
81 %> @param option [char] | The option for return value. Can be 'raw',
82 %> 'index', 'babel' or empty.
83 %>
84 %> @retval x [numeric array] | An array of wavelengths or wavelength
85 %> indexes
86 %======================================================================
87 function [x] = GetWavelengths(m, option)
88 % GetWavelengths returns the wavelengths of the hyperspectral image
89 %
90 % Depending on the option, it either returns the wavelengths or the
91 % wavelength indexes of the spectral image.
92 %
93 % @b Usage
94 %
95 % @code
96 % x = hsiUtility.GetWavelengths(m) returns wavelengths as a vector of wavelengths
97 %
98 % x = hsiUtility.GetWavelengths(m, 'raw') returns wavelengths as a vector of wavelengths
99 %
100 % x = hsiUtility.GetWavelengths(m, 'index') returns indexes respective to selected wavelengths
101 %
102 % x = hsiUtility.GetWavelengths(m, 'babel') returns indexes respective to selected wavelengths for babel standard spectra
103 % @endcode
104 %
105 % @param m [int] | The total number of spectral channels
106 % @param option [char] | The option for return value. Can be 'raw',
107 % 'index', 'babel' or empty.
108 %
109 % @retval x [numeric array] | An array of wavelengths or wavelength
110 % indexes
111
112 if nargin < 2
113 option = 'raw';
114 end
115
116 switch option
117 case 'raw'
118 splitWavelength = config.GetSetting('splitWavelength');
119 if m == 401
120 x = 380:780;
121 elseif m == 36
122 x = 380:10:730;
123 elseif m == 32
124 x = 420:10:730; % range [420,730];
125 elseif m == 17
126 x = 380:10:splitWavelength;
127 elseif m == 19
128 x = (splitWavelength + 1):10:730;
129 elseif m == 161
130 x = 380:splitWavelength;
131 elseif m == 240
132 x = (splitWavelength + 1):780;
133 elseif m == 311
134 x = 420:730;
135 else
136 error('Unsupported wavelength range');
137 end
138
139 case 'index'
140 x = hsiUtility.GetWavelengths(m, 'raw');
141 x = x - 380 + 1;
142
143 case 'babel'
144 if m == 36
145 x = 1:36;
146 elseif m == 17
147 x = 1:17;
148 elseif m == 19
149 x = 18:36;
150 else
151 error('Unsupported wavelengths range');
152 end
153
154 otherwise
155 error('Unsupported option.')
156
157 end
158
159 end

321 %% Setup
322 disp('Initializing [ExportH5Dataset]...');
323
324 fileName = commonUtility.GetFilename('output', ...
325 fullfile(config.GetSetting('datasetsFolderName'), strcat('hsi_', config.GetSetting('dataset'), '_full')), 'h5');

337 %% load HSI from .mat file

340 if (hsi.IsHsi(spectralData))
341
342 dataValue = spectralData.Value;
343 dataMask = uint8(spectralData.FgMask);
344 label = labelInfo.Labels;

347 end
348 sampleID = str2num(spectralData.SampleID);
349 targetID = str2num(spectralData.ID);
350
351 curName = strcat('/hsi/sample', targetName);
352 h5create(fileName, curName, size(dataValue));
353 h5write(fileName, curName, dataValue);
354
355 curName = strcat('/mask/sample', targetName);
356 h5create(fileName, curName, size(dataMask));
357 h5write(fileName, curName, dataMask);

359 curName = strcat('/label/sample', targetName);
360 h5create(fileName, curName, size(label));
361 h5write(fileName, curName, label);
362
363 curName = strcat('/sampleID/sample', targetName);
364 h5create(fileName, curName, size(sampleID));
365 h5write(fileName, curName, sampleID);
366
367 curName = strcat('/targetID/sample', targetName);

369 h5write(fileName, curName, targetID);

371 else
372 dataValue = spectralData;
