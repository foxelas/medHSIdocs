30tmp = delimread(inputSettingsFile, {', '}, 'raw');
31for i = 1:length(tmp.raw)
32
33 row = tmp.raw{i};
34 if ~((contains(row, '[') && contains(row, ']')) || contains(row, '#'))
35 % Ignore section header and commented out lines
36 parts = strsplit(row, '=');
37 varName = strtrim(parts{1});
38 rawValue = [];
39 if length(parts) > 1
40 rawValue = strtrim(parts{2});
41 end
42
43 if isempty(rawValue)
44 varValue = [];
45 switch varName
46 case 'importDir'
47 varValue = fullfile(dataDir, 'import\');
48 case 'matDir'
49 varValue = fullfile(outputDir, 'matfiles', 'hsi\');
50 case 'paramDir'
51 varValue = fullfile(dataDir, 'parameters\');
52 end
53
54 elseif strcmpi(rawValue, 'true') || strcmpi(rawValue, 'false') % logical type
55 varValue = strcmpi(rawValue, 'true');
56 elseif (contains(row, '{') && contains(row, '}')) % array type
57 varValue = GetArray(rawValue);
58 else % numeric or string type
59 varValue = GetValueWithType(rawValue);
60 end
61
62 eval([varName, '=', 'varValue', ';']);
63 end
64end
65
66fprintf('Data directory is set to %s.\n', dataDir);
67fprintf('Import directory is set to %s.\n', importDir);
68fprintf('Output directory is set to %s.\n', outputDir);
69fprintf('Parameter directory is set to %s.\n', paramDir);
70fprintf('Matfile directory is set to %s.\n', matDir);
71
72clear parts row varName rawValue varValue i tmp;
73settingsFile = strrep(inputSettingsFile, '.ini', '.mat');
74save(settingsFile);
75fprintf('\nSettings loaded from %s \n and saved in %s.\n\n', inputSettingsFile, settingsFile);
